439 pcl::PointCloud<pcl::PointXYZRGB> openni2_grabber::getRGBPointCloud()
440 {
441  if(!isSyncReg) std::cout<<"Depth to color Registration/ Depth-Color sync is not enabled..\n";
442  pcl::PointCloud<pcl::PointXYZRGB> cloud;
443 
444  float px,py,pz;
445 
446  DepthPixel *depthPixels,dp;
447  RGB888Pixel *colorPixels,cp;
448 
449  depthPixels = (DepthPixel*)((char*)getDepthFrame().getData());
450  colorPixels = (RGB888Pixel*)((char*)getColorFrame().getData());
451 
452  cloud.width = depthFrame.getWidth();
453  cloud.height = depthFrame.getHeight();
454  cloud.resize(cloud.width * cloud.height);
455 
456  for(int y=0;y<depthFrame.getHeight();y++)
457  {
458  for(int x=0;x<depthFrame.getWidth();x++)
459  {
460  dp = *depthPixels;
461  cp = *colorPixels;
462  status = CoordinateConverter::convertDepthToWorld(depthSensor,x,y,dp,&px,&py,&pz);
463  if(!handlestatus(status))
464  {
465  ReadLastCharOfLine();
466  exit(EXIT_FAILURE);
467  }
468  cloud.points[x + (depthFrame.getStrideInBytes() * y)/2].x = px;
469  cloud.points[x + (depthFrame.getStrideInBytes() * y)/2].y = py;
470  cloud.points[x + (depthFrame.getStrideInBytes() * y)/2].z = pz;
471  cloud.points[x + (depthFrame.getStrideInBytes() * y)/2].r = cp.r;
472  cloud.points[x + (depthFrame.getStrideInBytes() * y)/2].g = cp.g;
473  cloud.points[x + (depthFrame.getStrideInBytes() * y)/2].b = cp.b;
474 
475  depthPixels++;
476  colorPixels++;
477  }
478  }
479  return cloud;
